class TestWhile
{
    void main () {
        Int a;
        Int b;
        
        println("Loop with comparison with constant");
        a = 10;
        while(a > 0) {
            println(a);
            a = a - 1;
        }
        
        println("Loop with comparison between variables");
        a = 10;
        b = 0;
        while(a > b) {
            println(b);
            b = b + 1;
        }
        
        println("Testing nested loop");
        a = 0;
        while(a < 10) {
            b = 0;
            while(b < 10) {
                println(a + b);
                b = b + 1;
            }
            a = a + 1;
        }
        
        while(false) {
            println("this shouldn't be printed");
        }
        
        while(true) {
            println("this should print forever");
        }
    }
}
