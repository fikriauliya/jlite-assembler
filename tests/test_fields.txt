class FieldAccessMain {
	void main () {
		Int a;
		Int b;
		Int temp;
		
		FieldAccess f;
		f = new FieldAccess();
		
		f.a = 7;  // Field write
		
		b = f.func(5); // Function call. Result should be 18
		
		temp = (f.a + b) *2; // Field read. Result should be (18 + 7) *2 = 50
		
		println (temp); // Should output 36
	}
}

class FieldAccess{ 
	Int a;
	Int func (Int x) {
		return (6 + x + a);
    }
}
